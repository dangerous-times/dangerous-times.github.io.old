create procedure setPopRate()
    BEGIN
    # Divide by these variables to calculate the population rate 
    SET @usPop = 328239523/1000000;   
    SET @whitePop = 0.601*@usPop;
    SET @blackPop = 0.133*@usPop;
    SET @blackMalePop = 0.492*@blackPop;
    SET @blackBoyPop = 0.223*@blackMalePop;
    SET @blackMenPop = @blackMalePop-@blackBoyPop;
    SET @hispanicPop = 0.183*@usPop;
    SET @otherPop = @usPop-@whitePop-@blackPop-@hispanicPop;
    END$$

  left join (SELECT agency, state, race, count(id) as cnt FROM mappingPoliceViolence_org 
    group by agency, state, race) as t1 
    on t0.agency = t1.agency and t0.state = t1.state 
    and t0.race = t1.race

    and (t0.race='black' or t0.race='white') and cnt=1
  group by t0.race
